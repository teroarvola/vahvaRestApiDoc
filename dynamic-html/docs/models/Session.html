Session

  • sessionId : TechnicalId
  • plannedStart : String
    Planned start of the session
  • actualStart : String
    Actual start of the session
  • place : String
    Place of the session
  • type : SessionType
  • status : SessionStatus
  • extra : Boolean
    Indicated wheter session is scheduled or and extra session
  • sessionIdentifier : String
    Session identifier
  • changedTime : String
    Last update timestamp
  • removed : Boolean
    True, is the session has been deleted, false otherwise
